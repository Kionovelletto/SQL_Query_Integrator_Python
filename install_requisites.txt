# Conectar ao SqlServer e criar:
# SINTAXE: /opt/mssql-tools/bin/sqlcmd -S localhost -U sa -P SqlServer2022!

CREATE DATABASE caio_db;
GO
USE caio_db;
GO
CREATE TABLE Vendas (
    id_venda int, 
    cliente varchar(255), 
    produto varchar(255), 
    data_venda date, 
    preco decimal(12, 2), 
    quantidade int,
    )
GO

# Inserir os dados:
# SINTAXE: /opt/mssql-tools/bin/sqlcmd -S localhost -U sa -P SqlServer2022!

USE caio_db;
INSERT INTO Vendas(id_venda, cliente, produto, data_venda, preco, quantidade)
VALUES
    (1, 'Caio', 'Computer', GetDate(), 7680, 1)
GO

# Consultar os dados inseridos:
# SINTAXE: /opt/mssql-tools/bin/sqlcmd -S localhost -U sa -P SqlServer2022!

USE caio_db;
SELECT * FROM Vendas;
GO
